# 마이크로 비트 시리얼 통신(UART) 수신

*수정중

```
# Imports go at the top
from microbit import *

uart.init(9600)
# Code in a 'while True:' loop repeats forever
while True:
    if uart.any():
        mes=uart.read()
        if mes=='0':
            
            display.show(Image.HEART)
            sleep(1000)
            
        if mes=='1':
            
            display.show(Image.CONFUSED)
            sleep(1000)   
   
```
