from microbit import * 

blank = []
numbers = ['one', 'two', 'three', 'four', 'five']
print(blank)
print(numbers) #display.show(), display.scroll()은 전체 리스트 값을 인자로 받지 못함.

# display.show(numbers) ->오류

display.show(str(numbers))
